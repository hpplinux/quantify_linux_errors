\begin{frame}[t,fragile]{Kconfig - Grammar}
    \begin{Verbatim}
K ->      "config" ID TYPE OPT*
        | "if" ID K "endif"
        | "choice" K+ "endchoice"

TYPE ->   "bool" | "tristate" | "string" | "int"

OPT  ->   "default" STRING
        | "range" INT
        | "depends on" ID
        | "select" ID
    \end{Verbatim}
\end{frame}
